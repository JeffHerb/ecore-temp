    function openExternalLink(strURL) {

    function buildExternalLinkUrl(strURL) {

        var parameterMappingStr = strURL.substring(strURL.indexOf("?") + 1);
        var parameterMappings = parameterMappingStr.match(/{(\w|\s)*}/g);

        for (i = 0; i < parameterMappings.length; i++) {
            var fldName = parameterMappings[i].substring(1, parameterMappings[i].length - 1);
            fldName = fldName.replace(/^\s+/g, '').replace(/\s+$/g, '');
            var fldRef = eval("document.elForm." + fldName);
            var fldValue = "";
            if (fldRef != null && fldRef != 'undefined') {
                fldValue = fldRef.value;
            }
            parameterMappingStr = parameterMappingStr.replace(/{(\w|\s)*}/, fldValue);
        }
        return (strURL.substring(0, strURL.indexOf("?") + 1) + parameterMappingStr);
    }

    /*
     * Create form
     *  - Use previously gathered parameters and values
     *  - One line of HTML for each set of data
     *  - POST method
     *
     */
    function createDynamicForm(targetFormName, strUrl) {

        var input;
        var targetForm = document.createElement('form');
        var frmAction = strUrl.substring(0, strUrl.indexOf("?"));
        var urlParamList = strUrl.substring(strUrl.indexOf("?") + 1).split("&");

        // Reset & empty the form object
        while (targetForm.hasChildNodes()) {
            targetForm.removeChild(targetForm.firstChild);
        }

        // Set up the form
        targetForm.setAttribute("name", targetFormName);
        targetForm.setAttribute("method", 'post');
        targetForm.setAttribute("target", 'newwindow');
        targetForm.setAttribute("action", frmAction);

        //create <input>s and set attributes
        if (urlParamList != null && urlParamList.length >= 1) {
            for (i = 0; i < urlParamList.length; i++) {
                var paramMapping = urlParamList[i];
                var paramName = paramMapping.substring(0, paramMapping.indexOf('='));
                var paramVal = paramMapping.substring(paramMapping.indexOf('=') + 1);

                if (paramName != null) {
                    paramName = paramName.replace(/^\s+/g, '').replace(/\s+$/g, '');
                    if (paramName.length <= 0) continue;
                    input = document.createElement('input');
                    input.setAttribute('name', paramName);
                    input.setAttribute('value', paramVal);
                    input.setAttribute('type', 'hidden');
                    targetForm.appendChild(input);
                    input = null;
                }
            }
        }

        //append the form to the document
        document.body.appendChild(targetForm);
        return targetForm;
    }

    function Map() {
        this.aname = new Array();
        this.avalue = new Array();
        this.asize = -1;

        /**
         * adds a name value pair
         */
        this.put = function(nkey, nvalue) {
            //if this value is already in here
            //remove it
            if (this.contains(nkey)) {
                //remove the old value
                this.remove(nkey);
            }
            //increase to add to the end
            this.asize++;

            this.aname[this.asize] = nkey;
            this.avalue[this.asize] = nvalue;
        }

        /**
         * gets all the keys from this map
         * as an array
         */
        this.getKeysAsArray = function() {
            return this.aname;
        }

        /**
         * Gets a value from a key
         */
        this.get = function(keyname) {
            var m_x = 0;
            for (; m_x < this.size(); m_x++) {
                if (this.aname[m_x] == keyname) {
                    return this.avalue[m_x];
                }
            }
            return null
        }

        /**
         * Removes the first occurrence in this map of the specified key
         */
        this.remove = function(key) {
            var idx = this.indexOf(key);
            this.removeByIndex(idx);
        };

        /**
         * see if keyname is already in this map
         * lame algorithm
         */
        this.contains = function(keyname) {
            var s = this.size();
            var m_i = 0;
            for (; m_i < s; m_i++) {
                if (this.aname[m_i] == keyname) {
                    return true;
                }
            }

            return false;
        }

        /**
         * Returns the index the key or -1 if this map does not contain
         * the key
         * int
         */
        this.indexOf = function(key) {
            var l_x = 0;
            for (; l_x < this.size(); l_x++) {
                if (this.aname[l_x] == key) {
                    return l_x;
                }
            }

            return -1;
        };

        /**
         * Removes the element at the specified position in this list (optional operation).
         * Object
         */
        this.removeByIndex = function(index) {
            if (index > this.asize)
                throw new Error(
                    "Map remove index is > size can't remove key/value"
                );
            //this is IE 5.5 and higher... write my own?
            this.aname.splice(index, 1);
            this.avalue.splice(index, 1);
            this.asize--;
        };

        /**
         * gets the size of this map object 2
         */
        this.size = function() {
            return this.asize + 1;
        }

        /**
         * the ol'toString
         */
        this.toString = function() {
            var str = "[";

            if (this.size() > 0) {
                var m_x = 0;
                for (; m_x < this.size(); m_x++) {
                    str += this.aname[m_x] + "=" + this.avalue[m_x] + ",";
                }
                //remove that last comma
                str = str.substring(0, str.length - 1);

            } else {
                str += "empty"
            }

            str += "]";

            return str;
        }
    }
